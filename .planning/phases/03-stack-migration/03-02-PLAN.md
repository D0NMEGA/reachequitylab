import { useState } from "react";

          className={`items-center gap-2 ${

              onClick={() => setMenuOpen(false)}

              {link.label}
            </Link>
          ))}
          <Link
            href="/contact"
            className={`no-underline font-semibold px-3 py-2 rounded-lg transition-colors duration-200 ${
              pathname === "/contact"
                ? "bg-brand-teal-500 text-white"
                : "bg-brand-blue-900 text-white hover:bg-brand-teal-500"
            }`}
            onClick={() => setMenuOpen(false)}
          >
            Contact Us
          </Link>
        </nav>
        <button
          className="md:hidden border border-border bg-transparent text-brand-blue-900 text-[1.2rem] px-2.5 py-1.5 rounded-[10px]"
